131  template <typename slice_type>
132  [[nodiscard]] slice_type subSlice(size_t begin, size_t end) const {
133  this->rangeCheck(begin);
134  // end == size() is a legal value, since end is the first
135  // element beyond the slice
136  // end == 0 is not a legal value (subtraction will underflow and
137  // throw an exception)
138  this->rangeCheck(end - 1);
139  // additions are safe, begin and end are smaller than size()
140  const size_t new_begin = begin + this->begin_;
141  const size_t new_end = this->begin_ + end;
142  if (new_end > this->end_) {
143  throw std::out_of_range("Invalid input parameters to slice");
144  }
145  return slice_type(storage_.data_, new_begin, new_end);
146  }
