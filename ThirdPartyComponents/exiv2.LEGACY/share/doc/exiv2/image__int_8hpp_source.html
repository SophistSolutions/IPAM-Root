48 template <typename T>
49 std::ostream& operator<<(std::ostream& stream, const binaryToStringHelper<T>& binToStr) {
50  for (size_t i = 0; i < binToStr.buf_.size(); ++i) {
51  auto c = static_cast<int>(binToStr.buf_.at(i));
52  const bool bTrailingNull = c == 0 && i == binToStr.buf_.size() - 1;
53  if (!bTrailingNull) {
54  if (c < ' ' || c >= 127) {
55  c = '.';
56  }
57  stream.put(static_cast<char>(c));
58  }
59  }
60  return stream;
61 }
