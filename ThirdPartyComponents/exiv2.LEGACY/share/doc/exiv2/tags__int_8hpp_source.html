207 template <size_t N, const TagDetailsBitmask (&array)[N]>
208 std::ostream& printTagBitmask(std::ostream& os, const Value& value, const ExifData*) {
209  const auto val = value.toUint32();
210  if (val == 0 && N > 0) {
211  auto [mask, label] = *array;
212  if (mask == 0)
213  return os << exvGettext(label);
214  }
215  bool sep = false;
216  for (size_t i = 0; i < N; ++i) {
217  // *& acrobatics is a workaround for a MSVC 7.1 bug
218  auto [mask, label] = *(array + i);
219 
220  if (val & mask) {
221  if (sep) {
222  os << ", " << exvGettext(label);
223  } else {
224  os << exvGettext(label);
225  sep = true;
226  }
227  }
228  }
229  return os;
230 }

241 template <size_t N, const TagDetailsBitlistSorted (&array)[N]>
242 std::ostream& printTagBitlistAllLE(std::ostream& os, const Value& value, const ExifData*) {
243  if (N == 0)
244  throw Error(ErrorCode::kerErrorMessage, std::string("Passed zero length TagDetailsBitlistSorted"));
245 
246  uint32_t vN = 0;
247  uint32_t currentVNBit = 0;
248  size_t lastArrayPos = 0; // Prevents unneeded searching of array
249  constexpr auto maxArrayBit = (array + N - 1)->first;
250  auto allVNZero = true;
251  auto useSep = false;
252 
253  // For each value
254  for (size_t i = 0; i < value.count(); i++) {
255  vN = value.toUint32(i);
256  if (vN == 0) { // If all bits zero, then nothing to process
257  currentVNBit += 8;
258  continue;
259  }
260  allVNZero = false;
261  // Cycle through every bit in that byte
262  for (auto j = 0; j < 8; j++, currentVNBit++) {
263  if ((vN >> j & 0x01) == 0) { // If bit not set, then nothing to process
264  continue;
265  }
266  if (currentVNBit > maxArrayBit) { // If beyond array values, output unknown index
267  os << ", [" << currentVNBit << "]";
268  continue;
269  }
270 
271  // Check to see if the numbered bit is found in the array
272  for (size_t k = lastArrayPos; k < N; ++k) {
273  // *& acrobatics is a workaround for a MSVC 7.1 bug
274  auto [bit, label] = *(array + k);
275 
276  if (currentVNBit == bit) {
277  lastArrayPos = k;
278  if (useSep) {
279  os << ", " << exvGettext(label);
280  } else {
281  os << exvGettext(label);
282  useSep = true;
283  }
284  break;
285  }
286  }
287  }
288  }
289  if (allVNZero)
290  os << exvGettext("None");
291  return os;
292 }
