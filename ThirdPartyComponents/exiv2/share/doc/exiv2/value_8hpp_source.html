817 struct LangAltValueComparator {
819  bool operator()(const std::string& str1, const std::string& str2) const {
820  int result = str1.size() < str2.size() ? 1 : str1.size() > str2.size() ? -1 : 0;
821  if (result == 0) {
822  for (auto c1 = str1.begin(), c2 = str2.begin(); result == 0 && c1 != str1.end(); ++c1, ++c2) {
823  result = tolower(*c1) < tolower(*c2) ? 1 : tolower(*c1) > tolower(*c2) ? -1 : 0;
824  }
825  }
826  return result < 0;
827  }

1253  const auto v = value_.at(n);

1262  template <typename I>
1263  inline I rational_to_integer_helper(size_t n) const {
1264  auto a = value_.at(n).first;
1265  auto b = value_.at(n).second;
1266 
1267  // Protect against divide-by-zero.
1268  if (b <= 0) {
1269  return 0;
1270  }
1271 
1272  // Check for integer overflow.
1273  if (std::is_signed<I>::value == std::is_signed<decltype(a)>::value) {
1274  // conversion does not change sign
1275  const auto imin = std::numeric_limits<I>::min();
1276  const auto imax = std::numeric_limits<I>::max();
1277  if (imax < b || a < imin || imax < a) {
1278  return 0;
1279  }
1280  } else if (std::is_signed<I>::value) {
1281  // conversion is from unsigned to signed
1282  const auto imax = typename std::make_unsigned<I>::type(std::numeric_limits<I>::max());
1283  if (imax < b || imax < a) {
1284  return 0;
1285  }
1286  } else {
1287  // conversion is from signed to unsigned
1288  const auto imax = std::numeric_limits<I>::max();
1289  if (a < 0) {
1290  return 0;
1291  }
1292  // Inputs are not negative so convert them to unsigned.
1293  const auto a_u = typename std::make_unsigned<decltype(a)>::type(a);
1294  const auto b_u = typename std::make_unsigned<decltype(b)>::type(b);
1295  if (imax < b_u || imax < a_u) {
1296  return 0;
1297  }
1298  }
1299 
1300  return static_cast<I>(a) / static_cast<I>(b);
1301  }
